        // --- PRO AUDIO ENGINE (DOSYA GEREKTİRMEZ) ---
        // Bu kod tarayıcının kendi sentezleyicisini kullanarak retro bir ses üretir.
        
        const enterScreen = document.getElementById('enter-screen');
        const mainContent = document.getElementById('main-content');
        const musicBtn = document.getElementById('music-btn');
        const musicIcon = document.getElementById('music-icon');
        
        let audioCtx;
        let isPlaying = false;
        let noiseNode, gainNode, droneOsc, droneGain;

        function initAudio() {
            const AudioContext = window.AudioContext || window.webkitAudioContext;
            audioCtx = new AudioContext();

            // 1. KATMAN: NOISE (PLAK CIZIRTISI)
            const bufferSize = 2 * audioCtx.sampleRate;
            const noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
            const output = noiseBuffer.getChannelData(0);
            for (let i = 0; i < bufferSize; i++) {
                // Brown noise (daha yumuşak ve retro)
                const white = Math.random() * 2 - 1;
                output[i] = (lastOut + (0.02 * white)) / 1.02;
                lastOut = output[i];
                output[i] *= 3.5; 
            }
            noiseNode = audioCtx.createBufferSource();
            noiseNode.buffer = noiseBuffer;
            noiseNode.loop = true;
            gainNode = audioCtx.createGain();
            gainNode.gain.value = 0.08; // Cızırtı ses seviyesi
            noiseNode.connect(gainNode);
            gainNode.connect(audioCtx.destination);

            // 2. KATMAN: DRONE (GİZEMLİ BAS)
            droneOsc = audioCtx.createOscillator();
            droneOsc.type = 'triangle';
            droneOsc.frequency.value = 55; // Düşük frekans (bas)
            droneGain = audioCtx.createGain();
            droneGain.gain.value = 0.1; // Drone ses seviyesi
            
            // Hafif dalgalanma (LFO)
            const lfo = audioCtx.createOscillator();
            lfo.type = 'sine';
            lfo.frequency.value = 0.2;
            const lfoGain = audioCtx.createGain();
            lfoGain.gain.value = 10;
            lfo.connect(lfoGain);
            lfoGain.connect(droneOsc.frequency);
            lfo.start();

            droneOsc.connect(droneGain);
            droneGain.connect(audioCtx.destination);
            
            noiseNode.start();
            droneOsc.start();
            isPlaying = true;
            updateMusicIcon();
        }
        
        let lastOut = 0;

        function toggleAudio() {
            if (!audioCtx) {
                initAudio();
                return;
            }
            if (audioCtx.state === 'running') {
                audioCtx.suspend();
                isPlaying = false;
            } else if (audioCtx.state === 'suspended') {
                audioCtx.resume();
                isPlaying = true;
            }
            updateMusicIcon();
        }

        function updateMusicIcon() {
            if (isPlaying) {
                musicIcon.classList.remove('fa-play');
                musicIcon.classList.add('fa-compact-disc', 'spin-record');
                musicBtn.classList.remove('pulse-btn');
            } else {
                musicIcon.classList.remove('fa-compact-disc', 'spin-record');
                musicIcon.classList.add('fa-play');
                musicBtn.classList.add('pulse-btn');
            }
        }

        // --- GİRİŞ VE SCROLL MANTIĞI ---

        enterScreen.addEventListener('click', function() {
            initAudio(); // Sesi başlat
            
            enterScreen.style.opacity = '0';
            setTimeout(() => {
                enterScreen.style.display = 'none';
            }, 1000);

            mainContent.style.display = 'block';
            setTimeout(() => {
                mainContent.style.opacity = '1';
                mainContent.style.transform = 'translateY(0)';
            }, 100);
        });

        musicBtn.addEventListener('click', toggleAudio);

        // Gelişmiş Scroll Gözlemcisi (Intersection Observer)
        // Kartlar ekrana girdiğinde "visible" class'ını ekler
        const observerOptions = {
            threshold: 0.15, // Kartın %15'i görünür olduğunda tetikle
            rootMargin: "0px 0px -50px 0px"
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                    // Girdikten sonra gözlemlemeyi bırak (performans için)
                    observer.unobserve(entry.target); 
                }
            });
        }, observerOptions);

        document.querySelectorAll('.card').forEach(card => {
            observer.observe(card);
        });

    </script>
